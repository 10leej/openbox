-- Keybinding  
META=Super key  
META+W = Close Window  
META+space = showdesktop toggle

META+R = Launcher (dmenu)  
META_T = Terminal (lxterminal)
META+B = Browser (firefox)
META+H = File Manager (thunar)

-- Window Tiling  
+-----------+----------+
| META+Home | META+PgUp |
+-----------+----------+ 
| META+End  | META+PgDn |
+-----------+----------+
META+Up = Maximize
META+Down = Unmaximize
META+Left = Tile Left
META+Right = Tile Right
